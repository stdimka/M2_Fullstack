Методы массивов в JavaScript

splice()

Синтаксис: array.splice(start, deleteCount, item1, ..., itemN)

Описание: Изменяет содержимое массива, удаляя или добавляя элементы.

let arr = [1, 2, 3, 4, 5];
arr.splice(2, 1, 99);
console.log(arr); // [1, 2, 99, 4, 5]

slice()

Синтаксис: array.slice(start, end)

Описание: Возвращает новый массив, содержащий копию части исходного массива.

let arr = [1, 2, 3, 4, 5];
let newArr = arr.slice(1, 4);
console.log(newArr); // [2, 3, 4]

concat()

Синтаксис: array.concat(arr2, arr3, ...)

Описание: Объединяет два или более массива, не изменяя исходные.

let arr1 = [1, 2];
let arr2 = [3, 4];
let arr3 = arr1.concat(arr2);
console.log(arr3); // [1, 2, 3, 4]

indexOf()

Синтаксис: array.indexOf(searchElement, fromIndex)

Описание: Возвращает первый индекс элемента в массиве или -1, если элемент не найден.

let arr = [10, 20, 30, 40];
console.log(arr.indexOf(30)); // 2

forEach()

Синтаксис: array.forEach(callback)

Описание: Выполняет функцию для каждого элемента массива.

let arr = [1, 2, 3];
arr.forEach(num => console.log(num * 2));

map()

Синтаксис: array.map(callback)

Описание: Создаёт новый массив, применяя функцию к каждому элементу.

let arr = [1, 2, 3];
let squared = arr.map(num => num ** 2);
console.log(squared); // [1, 4, 9]

filter()

Синтаксис: array.filter(callback)

Описание: Возвращает новый массив с элементами, которые соответствуют условию.

let arr = [1, 2, 3, 4];
let even = arr.filter(num => num % 2 === 0);
console.log(even); // [2, 4]

reduce()

Синтаксис: array.reduce(callback, initialValue)

Описание: Применяет функцию к каждому элементу, сводя массив к одному значению.

let arr = [1, 2, 3, 4];
let sum = arr.reduce((acc, num) => acc + num, 0);
console.log(sum); // 10

some()

Синтаксис: array.some(callback)

Описание: Проверяет, удовлетворяет ли хотя бы один элемент массива условию.

let arr = [1, 2, 3, 4];
console.log(arr.some(num => num > 3)); // true

every()

Синтаксис: array.every(callback)

Описание: Проверяет, удовлетворяют ли все элементы массива условию.

let arr = [2, 4, 6];
console.log(arr.every(num => num % 2 === 0)); // true

find()

Синтаксис: array.find(callback)

Описание: Возвращает первый элемент массива, удовлетворяющий условию.

let arr = [1, 2, 3, 4];
console.log(arr.find(num => num > 2)); // 3

findIndex()

Синтаксис: array.findIndex(callback)

Описание: Возвращает индекс первого элемента массива, удовлетворяющего условию.

let arr = [1, 2, 3, 4];
console.log(arr.findIndex(num => num > 2)); // 2
